-- fetch all security groups
select * from dbo.di_info_securityGroup
where active = 1;
--order by securityGroupName;

-- grant workspace connect tab access
insert into dbo.di_info_workspaceConnectAccess (securityGroupId, isActive, creationDateTime, createdBy)
values ('your-azure-ad-group-object-id', 1, getutcdate(), 'system');

-- fetch groups that already have workspace connect access
select a.securityGroupId, sg.securityGroupName, a.isActive, a.creationDateTime, a.createdBy
from dbo.di_info_workspaceConnectAccess a
join dbo.di_info_securityGroup sg on sg.securityGroupId = a.securityGroupId
where a.isActive = 1;

-- check if one sg has access (same as api)
exec dbo.sel_workspaceConnectAccess @securityGroupId = 'your-azure-ad-group-object-id';

-- add sg to master (if missing)
insert into dbo.di_info_securityGroup (securityGroupId, securityGroupName, active, creationDateTime)
values ('your-azure-ad-group-object-id', N'your group name', 1, getutcdate());
