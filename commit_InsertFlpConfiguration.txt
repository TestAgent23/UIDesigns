try
                    {
                        (bool success, string uploadedId) = await UploadedFileDetails(flpConfigurationId, item.Name, processTypeId);
                        if (success)
                        {
                            sharePointFiles.Add(new SharePointFileLocation
                            {
                                UploadedId = uploadedId,
                                FileUrl = item.Path ?? item.WebUrl ?? item.Name,
                                FileSize = item.Size
                            });
                        }
                        else
                        {
                            _logger.LogWarning(
                                "SharePoint file {FileName} found in {BrowsePath} but commit_fileUpload failed for {FlpConfigurationId}",
                                item.Name, browsePath, flpConfigurationId);
                        }
                    }
                    catch (Exception ex)
                    {
                        _logger.LogError(ex,
                            "Failed to register SharePoint file {FileName} for {FlpConfigurationId}",
                            item.Name, flpConfigurationId);
                    }
